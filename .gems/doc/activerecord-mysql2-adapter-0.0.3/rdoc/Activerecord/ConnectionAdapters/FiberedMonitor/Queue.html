def wait(timeout)
  t = timeout || 5
  fiber = Fiber.current
  x = EM::Timer.new(t) do
    @queue.delete(fiber)
    fiber.resume(false)
  end
  @queue << fiber
  Fiber.yield.tap do
    x.cancel
  end
end
